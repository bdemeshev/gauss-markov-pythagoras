\begin{tikzpicture}[scale=3]

\tkzDefPoint(0.866,0.5){e1}
\tkzDefPoint(0.433,0.25){a1}
\tkzDefPoint(0.25,-0.433){a2}
\tkzDefPoint(0.75,-1.299){a}

\begin{pgfinterruptboundingbox} % remove blank space

\tkzDefLine[parallel=through a](O,a1);
\tkzGetPoint{p2}
\draw (p2) -- (a);

\tkzInterLL(p1,a1)(p2,a);
\tkzGetPoint{C}
\end{pgfinterruptboundingbox}

\fill[pattern=north west lines, pattern color=gray] (a1) -- (O) -- (a) -- (C);
