\documentclass{article}
%%%<
\usepackage{verbatim}
%%%>

\usepackage{tikz}   %TikZ is required for this to work.  Make sure this exists before the next line

\usepackage{tikz-3dplot} %requires 3dplot.sty to be in same directory, or in your LaTeX installation
\usepackage{tkz-euclide}
\usetkzobj{all}
\usetikzlibrary{calc}
\usepackage[active,tightpage]{preview}  %generates a tightly fitting border around the work
\PreviewEnvironment{tikzpicture}
\setlength\PreviewBorder{2mm}

\begin{document}
\begin{tikzpicture}[scale=3]

\tkzDefPoint(0,0){O}
\tkzDefPoint(1,0){e1}
\tkzDefPoint(0.866,0.5){e2} % (cos(30), sin(30))
\tkzDefPoint(1.75,0){x1}
\tkzDefPoint(0.6495,0.375){x2} % (cos(30)*3/4, sin(30)*3/4)

\draw (O) circle (1cm);
\draw [->] (O) -- (e1) node[anchor=north west]{$e_1$};
\draw [->] (O) -- (e2) node[anchor=south west]{$e_2$};
\draw [->] (O) -- (x1) node[anchor=north]{$x_1$};
\draw [->] (O) -- (x2) node[anchor=south east]{$x_2$};

\tkzMarkAngle[size=0.2cm](e1,O,e2)
\tkzLabelAngle[pos=0.3](e1,O,e2){$\alpha$}

\begin{pgfinterruptboundingbox} % remove blank space
\tkzDefLine[perpendicular=through x2](x1,O);
\tkzGetPoint{h1}
\tkzDefPointBy[projection=onto x1--O](h1)
\tkzGetPoint{hh1}
\tkzDefLine[parallel=through O](hh1,x2);
\tkzGetPoint{p1}
\draw[->] (O) -- (p1) node[anchor=east]{$\hat{u}_2$};

\tkzDefLine[perpendicular=through x1](x2,O);
\tkzGetPoint{h2}
\tkzDefPointBy[projection=onto x2--O](h2)
\tkzGetPoint{hh2}
\tkzDefLine[parallel=through O](hh2,x1);
\tkzGetPoint{p2}
\draw[->] (O) -- (p2) node[anchor=east]{$\hat{u}_1$};
\end{pgfinterruptboundingbox}

\end{tikzpicture}
\end{document}
